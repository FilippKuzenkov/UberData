CREATE OR REPLACE TABLE `uberdataproject-436710.uberdataset.analytics` AS (
SELECT 
    f.VendorID,
    f.trip_id,
    d.datetime_id,
    d.tpep_pickup_datetime,
    d.tpep_dropoff_datetime,
    p.passenger_count,
    t.trip_distance,
    r.rate_code_id,
    r.rate_code_name,
    pick.pickup_location_id,
    pick.pickup_latitude,
    pick.pickup_longitude,
    drop.dropoff_location_id,
    drop.dropoff_latitude,
    drop.dropoff_longitude,
    pay.payment_type_id,
    pay.payment_type_name,
    f.fare_amount,
    f.extra,
    f.mta_tax,
    f.tip_amount,
    f.tolls_amount,
    f.improvement_surcharge,
    f.total_amount
FROM 
    `uberdataproject-436710.uberdataset.fact_table` f
JOIN 
    `uberdataproject-436710.uberdataset.datetime_dim` d ON f.datetime_id = d.datetime_id
JOIN 
    `uberdataproject-436710.uberdataset.passenger_count_dim` p ON f.passenger_count_id = p.passenger_count_id  
JOIN 
    `uberdataproject-436710.uberdataset.trip_distance_dim` t ON f.trip_distance_id = t.trip_distance_id  
JOIN 
    `uberdataproject-436710.uberdataset.rate_code_dim` r ON f.rate_code_id = r.rate_code_id  
JOIN 
    `uberdataproject-436710.uberdataset.pickup_location_dim` pick ON f.pickup_location_id = pick.pickup_location_id
JOIN 
    `uberdataproject-436710.uberdataset.dropoff_location_dim` drop ON f.dropoff_location_id = drop.dropoff_location_id
JOIN 
    `uberdataproject-436710.uberdataset.payment_type_dim` pay ON f.payment_type_id = pay.payment_type_id
);
